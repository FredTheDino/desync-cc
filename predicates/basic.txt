predicate xorl_jz(reg_a: r32) {
    xorl %reg_a, %reg_a
    jz DESYNC
}

predicate xorq_jz(reg_a: r64) {
    xorq %reg_a, %reg_a
    jz DESYNC
}

predicate movl_jz(reg_a: r32) {
    movl $0, %reg_a
    jz DESYNC
}

predicate movq_jz(reg_a: r64) {
    movq $0, %reg_a
    jz DESYNC
}

predicate nop(reg_a: r32) {
    movl $0, %reg_a
}
